@RestController
@RequestMapping("/api/v3/global/drugs")
public class GlobalDrugController {
    private final GlobalDrugService service;

    public GlobalDrugController(GlobalDrugService service) {
        this.service = service;
    }

     @PostMapping("/add")
    public ResponseEntity<ApiResponseDto<GlobalDrugResponse>> create(
            @Valid @RequestBody CreateGlobalDrugRequest request) {

        GlobalDrug drug = service.createGlobalDrug(request);

        GlobalDrugResponse response = mapToResponse(drug);

        return ResponseEntity.status(HttpStatus.CREATED).body(
                ApiResponseDto.success("Global drug created successfully", response));
    }

    private GlobalDrugResponse mapToResponse(GlobalDrug drug) {
        GlobalDrugResponse r = new GlobalDrugResponse();
        r.setGlobalDrugId(drug.getGlobalDrugId());
        r.setBrandName(drug.getBrandName());
        r.setGenericName(drug.getGenericName());
        r.setManufacturer(drug.getManufacturer());
        r.setSystemOfMedicine(drug.getSystemOfMedicine());
        r.setDosageForm(drug.getDosageForm());
        r.setStrengthValue(drug.getStrengthValue());
        r.setStrengthUnit(drug.getStrengthUnit());
        r.setRouteOfAdministration(drug.getRouteOfAdministration());
        r.setBaseUnit(drug.getBaseUnit());
        r.setUnitsPerPack(drug.getUnitsPerPack());
        r.setLooseSaleAllowed(drug.getLooseSaleAllowed());
        r.setDrugSchedule(drug.getDrugSchedule());
        r.setPrescriptionRequired(drug.getPrescriptionRequired());
        r.setNarcoticDrug(drug.getNarcoticDrug());
        r.setRegulatoryAuthority(drug.getRegulatoryAuthority());
        r.setHsnCode(drug.getHsnCode());
        r.setVerified(drug.getVerified());
        r.setCreatedByChemistId(drug.getCreatedByChemistId());
        r.setCreatedAt(drug.getCreatedAt());
        r.setActive(drug.getActive());
        return r;
    }



}
