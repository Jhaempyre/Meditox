@RestController
@RequestMapping("/api/v3/chemist")
public class ChemistProductController {

        private final ChemistProductService chemistProductService;
        private final StockService stockService;

        public ChemistProductController(ChemistProductService chemistProductService, StockService stockService) {
                this.chemistProductService = chemistProductService;
                this.stockService = stockService;
        }

        @PostMapping("/{chemistId}/products")
        public ResponseEntity<ApiResponseDto<ChemistProductResponse>> addProductToCatalog(
                        @PathVariable Long chemistId,
                        @Valid @RequestBody AddProductToCatalogRequest request) {

                ChemistProductResponse response = chemistProductService.addProductToCatalog(chemistId, request);

                return ResponseEntity.status(HttpStatus.CREATED).body(
                                ApiResponseDto.success("Product added to chemist catalog successfully", response));
        }



//add to catalouge 

public class AddProductToCatalogRequest {

    @NotNull(message = "Product category is required")
    @JsonProperty("product_category")
    private ProductCategory productCategory;

    @NotNull(message = "Global product ID is required")
    @Positive(message = "Global product ID must be positive")
    @JsonProperty("global_product_id")
    private Long globalProductId;

    @NotNull(message = "Minimum stock level is required")
    @Min(value = 0, message = "Minimum stock level must be non-negative")
    @JsonProperty("minimum_stock_level")
    private Integer minimumStockLevel;

    @NotNull(message = "Maximum stock level is required")
    @Positive(message = "Maximum stock level must be positive")
    @JsonProperty("maximum_stock_level")
    private Integer maximumStockLevel;

    @NotNull(message = "Reorder quantity is required")
    @Positive(message = "Reorder quantity must be positive")
    @JsonProperty("reorder_quantity")
    private Integer reorderQuantity;
